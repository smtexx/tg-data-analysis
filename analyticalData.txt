Подробные данные телеграм каналов
База - telega.in

Структура: 
{
  id, - уникальный идентификатор категории
  name, - название категории
  channels: [
    {
      id, - уникальный идентификатор канала
      name, - название канала,
      description, - описание канала
      subscribers, - количество подписчиков
    }
  ],
  chats: [
    {
      id, - уникальный идентификатор чата
      name, - название канала,
      description, - описание канала
      subscribers, - количество подписчиков
    }
  ],
  CALCULATED: {
    totalGroups, - сумма каналов и чатов категории   
    totalMembers, - всего подписчиков в категории 
    chatStat: {
      total, - количество чатов в категории
      members, - суммарное количество участников чатов
      percentQty, - процент от общего количества групп
      percentMembers, - процент от общего количества участников
      avgMembers, - среднее количество участников
      distributionByMembersPercent: {
        20-, - процент групп имеющих не более 20 участников
        20-100, - процент групп имеющих от 20 до 100 участников
        100-500, - процент групп имеющих от 20 до 100 участников
        500-5k, - процент групп имеющих от 500 до 5 000 участников
        5k-50k, - процент групп имеющих от 5 000 до 50 000 участников
        50k-100k - процент групп имеющих от 50000 до 100000 участников
        100k-500k - процент групп имеющих от 100 000 до 500 000 участников
        500k+ - процент групп имеющих более 500 000 участников
      }
    channelStat: {
      total, - количество каналов в категории
      members, - суммарное количество подписчиков каналов
      percentQty, - процент от общего количества групп
      percentMembers, - процент от общего количества участников
      avgMembers, - среднее количество подписчиков
      distributionByMembersPercent: {
        20-, - процент групп имеющих не более 20 участников
        20-100, - процент групп имеющих от 20 до 100 участников
        100-500, - процент групп имеющих от 20 до 100 участников
        500-5k, - процент групп имеющих от 500 до 5 000 участников
        5k-50k, - процент групп имеющих от 5 000 до 50 000 участников
        50k-100k - процент групп имеющих от 50000 до 100000 участников
        100k-500k - процент групп имеющих от 100 000 до 500 000 участников
        500k+ - процент групп имеющих более 500 000 участников
      }
    }
  }
}
