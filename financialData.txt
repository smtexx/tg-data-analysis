Подробные данные телеграм каналов
База - telega.in

Структура: 
{
  categoryID: {
    title,
    keywords,
    channels: [
      {
        id,
        financial, - данные с telega.in
        stat, - данные с tgstat.ru
	CALCULATED: {
          totalIncome, - минимально заработано на канале
          monthIncome, - средний минимальный заработок в месяц
          incomePerMember, - минимально заработано с одного подписчика
          minAdsPrice, - минимальная стоимость рекламного поста
          maxAdsPrice, - максимальная стоимость рекламного поста
	  avgPostPerDay, - среднее количество постов в день
          category, - категория канала или чата
        }
      }
    ],
    CALCULATED: {
      // Деньги
      totalIncome, - всего заработано в категории
      totalOrders, - всего заказов рекламы в категории
      avgTotalIncome, - в среднем заработано на канал
      avgMonthIncome, - средний минимальный заработок в месяц на канал
      avgIncomePerMember, - в среднем заработано с одного подписчика
      avgMinAdsPrice, - средняя минимальная стоимость рекламного поста
      avgMaxAdsPrice, - средняя максимальная стоимость рекламного поста
      interestIndex, - индекс интереса: сумма всех просмотров карточки + сумма добавлений в избранное * 10 разделенная на количество каналов в категории
      incomeDistributionPercent: {
        noIncome, - процент каналов ничего не заработавших
        5-20k, - процент каналов заработавших от 5000 до 10000 руб
        20-50k - процент каналов заработавших от 20000 до 50000 руб
        50-100k - процент каналов заработавших от 50000 до 100000 руб
        100-500k - процент каналов заработавших от 100000 до 500000 руб
        500k+ - процент каналов заработавших более 500000 руб
      }

      // Статистика
      totalChannels, - всего каналов в категории
      totalMembers, - всего подписчиков в категории
      smallChannelsPercent, - процент каналов с количеством подписчиков до 100 человек
      avgMembersPerChannel, - в среднем подписчиков на канал  
      avgPostPerDay, - среднее количество постов в день 
      avgChannelAge, - средний возраст канала
      membersTrend: {
        week, - недельный тренд прироста подписчиков в категории
        month, - месячный тренд прироста подписчиков в категории
      }
      membersDistributionPercent: {
        publicChannel, - процент открытых каналов
        publicChat, - процент открытых чатов
        privateChannel, - процент приватных каналов
        privateChat, - процент приватных чатов
      }
      genderDistributionPercent: {
        male, - процент мужской аудитории
        female, - процент женской аудитории
      }      
    }
  }
}
